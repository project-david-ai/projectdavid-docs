\section{Supervisor Execution Loop (SEL)}

\begin{enumerate}

\item \texttt{update\_scratchpad([STRATEGY])}

\item \texttt{delegate\_research\_task()}

\item Upon worker return:
\[
read\_scratchpad()
\]

\item Evaluate:

\begin{itemize}
    \item If [unresolved] or [warning] entries exist — redelegate
    \item If all required fields = [verified] — synthesize
\end{itemize}

\item Final Synthesis:
\[
generate\_final\_response()
\]

\end{enumerate}
