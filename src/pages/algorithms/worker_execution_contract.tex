\section{Worker Execution Contract (WEC)}

\subsection*{Mandatory Ordering}

\begin{enumerate}
    \item \texttt{read\_scratchpad()} \textbf{and} \texttt{first\_research\_tool()}
    \item \texttt{append\_scratchpad([PENDING])}
    \item \texttt{perform\_research\_sequence()}
    \item \texttt{append\_scratchpad([verified] / [unresolved] / [warning] result)}
    \item \texttt{one\_line\_confirmation()}
\end{enumerate}

\subsection*{Hard Constraints}

\begin{itemize}
    \item No text before \texttt{append\_scratchpad}
    \item No reasoning-only completion
    \item No memory-only answer
\end{itemize}
